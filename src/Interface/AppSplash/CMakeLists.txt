##################################################
# Set sources
##################################################

SET(INTERFACE_APPSPLASH_SRCS
  AppSplash.cc
)

SET(INTERFACE_APPSPLASH_MOC_SRCS
  AppSplash.h
)

SET(INTERFACE_APPSPLASH_NO_MOC_SRCS
)

SET(INTERFACE_APPSPLASH_UI_SRCS
  AppSplash.ui
)

SET(INTERFACE_APPSPLASH_RCC_SRCS
)


##################################################
# Get Icon files
##################################################

QT4_ADD_RESOURCES(INTERFACE_APPSPLASH_QT_RCC_SRCS ${INTERFACE_APPSPLASH_RCC_SRCS})

##################################################
# Generate header out of UI code
##################################################

QT4_WRAP_UI(INTERFACE_APPSPLASH_QT_UI_SRCS ${INTERFACE_APPSPLASH_UI_SRCS})

##################################################
# Wrap QT code to expand all the moc code
##################################################

QT4_WRAP_CPP(INTERFACE_APPSPLASH_QT_MOC_SRCS ${INTERFACE_APPSPLASH_MOC_SRCS})

##################################################
# Ensure that we can find the files generated
# by the moc and ui builder
##################################################

INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})

##################################################
# Build the Components library
##################################################

CORE_ADD_LIBRARY(Interface_AppSplash 
            ${INTERFACE_APPSPLASH_SRCS} 
            ${INTERFACE_APPSPLASH_QT_UI_SRCS}
            ${INTERFACE_APPSPLASH_QT_RCC_SRCS}
            ${INTERFACE_APPSPLASH_QT_MOC_SRCS}
            ${INTERFACE_APPSPLASH_NO_MOC_SRCS} 
            ${INTERFACE_APPSPLASH_MOC_SRCS})

#ADD_DEPENDENCIES(Interface_AppSplash 
#			${INTERFACE_APPSPLASH_UI_SRCS}
#			${INTERFACE_APPSPLASH_RCC_SRCS} )	
            
TARGET_LINK_LIBRARIES(Interface_AppSplash
                      Core_Utils
                      Core_EventHandler
                      Core_Action
                      Core_Interface
                      Core_ActionManager
					  Application_ProjectManager
                      ${QT_LIBRARIES} 
                      ${SCI_BOOST_LIBRARY})
